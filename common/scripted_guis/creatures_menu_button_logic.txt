scripted_gui = { 
    topbar = {
        context_type = player_context
		window_name = "creatures_menu_button_container"
		
		visible = {
			is_ai = no
		}
		
		effects = {
			creatures_menu_button_click = {
				if = {
					limit = { NOT = { has_variable = pony_menu } }
					set_variable = { pony_menu = 1 }
				}
				else = {
					clear_variable = pony_menu
				}
				if = {
                    limit = {
                        NOT = {
                            has_variable = equipment_earthpony_checkbox_blank
                            has_variable = equipment_earthpony_checkbox_signed
                        }
                    }
                    set_variable = { equipment_earthpony_checkbox_blank = 1 }
                }
				if = {
                    limit = {
                        NOT = {
                            has_variable = equipment_unicorn_checkbox_blank
                            has_variable = equipment_unicorn_checkbox_signed
                        }
                    }
                    set_variable = { equipment_unicorn_checkbox_blank = 1 }
                }
				if = {
                    limit = {
                        NOT = {
                            has_variable = equipment_pegasus_checkbox_blank
                            has_variable = equipment_pegasus_checkbox_signed
                        }
                    }
                    set_variable = { equipment_pegasus_checkbox_blank = 1 }
                }
                if = {
                    limit = {
                        NOT = {
                            has_variable = equipment_zmih_checkbox_blank
                            has_variable = equipment_zmih_checkbox_signed
                        }
                    }
                    set_variable = { equipment_zmih_checkbox_blank = 1 }
                }
			}
		}
        
		triggers = {
        }
    }
}
