scripted_gui = {

    open_creatures_equipment_window_container_paws = {
    
        context_type = player_context
		
		window_name = "open_creatures_equipment_window_container_paws"
		
		visible = {
			is_ai = no
            has_variable = paws_menu
		}
		
		effects = {
        
			equipment_dogs_button_click = {
				ROOT = {
					set_variable = {
						var = race_equipment_type
						value = 4
					}
					set_variable = {
						var = race
						value = 12
					}

					log = "[?this.GetEquipmentType] equipment type for [?this.GetRace] race" #debug
					count_race_equipped = yes
					
					clear_variable = race_equipment_type
					clear_variable = race
				}
			}
			equipment_bears_button_click = {
				ROOT = {
					set_variable = {
						var = race_equipment_type
						value = 4
					}
					set_variable = {
						var = race
						value = 9
					}

					log = "[?this.GetEquipmentType] equipment type for [?this.GetRace] race" #debug
					count_race_equipped = yes
					
					clear_variable = race_equipment_type
					clear_variable = race
				}
			}

			# Selection bar
			button_pony_click = {
			add_to_variable = { pony_menu = 1 }
			clear_variable = paws_menu
			}
			button_crystal_click = {
			add_to_variable = { crystal_menu = 1 }
			clear_variable = paws_menu
            }
			button_hooves_click = {
			add_to_variable = { hooves_menu = 1 }
			clear_variable = paws_menu
			}
			button_claws_click = {
			add_to_variable = { claws_menu = 1 }
			clear_variable = paws_menu
			}
			button_yak_click = {
			add_to_variable = { yak_menu = 1 }
			clear_variable = paws_menu
			}
			button_penguins_click = {
			add_to_variable = { penguins_menu = 1 }
			clear_variable = paws_menu
			}
			button_magic_equipment_click = {
			add_to_variable = { magic_equipment_menu = 1 }
			clear_variable = paws_menu
			}
			equipment_dogs_checkbox_blank_click = {
			limit = { NOT = { has_variable = equipment_dogs_checkbox_signed } }
			set_variable = { equipment_dogs_checkbox_signed = 1 }
			clear_variable = equipment_dogs_checkbox_blank
			}
			equipment_dogs_checkbox_signed_click = {
			limit = { NOT = { has_variable = equipment_dogs_checkbox_blank } }
			set_variable = { equipment_dogs_checkbox_blank = 1 }
			clear_variable = equipment_dogs_checkbox_signed
			}
			equipment_bears_checkbox_blank_click = {
			limit = { NOT = { has_variable = equipment_crystal_unicorn_checkbox_signed } }
			set_variable = { equipment_bears_checkbox_signed = 1 }
			clear_variable = equipment_bears_checkbox_blank
			}
			equipment_bears_checkbox_signed_click = {
			limit = { NOT = { has_variable = equipment_bears_checkbox_blank } }
			set_variable = { equipment_bears_checkbox_blank = 1 }
			clear_variable = equipment_bears_checkbox_signed
			}
			button_close_click = {            
				clear_variable = paws_menu
			}
		}
        
		triggers = {
			equipment_dogs_checkbox_blank_visible = {
			has_variable = equipment_dogs_checkbox_blank
			}
			equipment_dogs_checkbox_signed_visible = {
			has_variable = equipment_dogs_checkbox_signed
			}
			equipment_bears_checkbox_blank_visible = {
			has_variable = equipment_bears_checkbox_blank
			}
			equipment_bears_checkbox_signed_visible = {
			has_variable = equipment_bears_checkbox_signed
			}
        }
    }
}
