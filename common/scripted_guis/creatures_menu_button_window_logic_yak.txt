scripted_gui = {

    open_creatures_equipment_window_container_yak = {
    
        context_type = player_context
		
		window_name = "open_creatures_equipment_window_container_yak"
		
		visible = {
			is_ai = no
        	has_variable = yak_menu
		}
		
		effects = {
			equipment_yak_button_click = {
				ROOT = {
					set_variable = {
						var = race_equipment_type
						value = 5
					}
					set_variable = {
						var = race
						value = 12
					}

					log = "[?this.GetEquipmentType] equipment type for [?this.GetRace] race" #debug
					count_race_equipped = yes
					
					clear_variable = race_equipment_type
					clear_variable = race
				}
			}

			# Selection bar
			button_pony_click = {
			add_to_variable = { pony_menu = 1 }
			clear_variable = yak_menu
			}
			button_crystal_click = {
			add_to_variable = { crystal_menu = 1 }
			clear_variable = yak_menu
            }
			button_hooves_click = {
			add_to_variable = { hooves_menu = 1 }
			clear_variable = yak_menu
			}
			button_claws_click = {
			add_to_variable = { claws_menu = 1 }
			clear_variable = yak_menu
			}
			button_paws_click = {
			add_to_variable = { paws_menu = 1 }
			clear_variable = yak_menu
			}
			button_penguins_click = {
			add_to_variable = { penguins_menu = 1 }
			clear_variable = yak_menu
			}
			button_magic_equipment_click = {
			add_to_variable = { magic_equipment_menu = 1 }
			clear_variable = yak_menu
			}
			button_close_click = {            
				clear_variable = yak_menu
			}
		}
        
		triggers = {
        }
    }
